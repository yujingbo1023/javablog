    # 2，配置式路由

    在config目录下面创建一个routes.ts文件，在这里配置路由，如下：

    const routes = [
    {
        path: "/",
        // 自动会去pages目录下查找，也可以也指定目录
        // component:'@/a/b'
        component: 'index',
        title:"首页"
    },{
        path: "/count",
        component: 'count',
        title:"计数器"
    },{
        path: "/user",
        component: 'user',
        title:"个人中心",
        routes:[
            {
                path: "/user",
                redirect: '/user/order',
            },
            {
                path: "/user/order",
                component: 'user/order',
                title:"订单页"
            },
            {
                path: "/user/profile",
                component: 'user/profile',
                title:"个人信息"
            }
        ]
    },{
        path: "*",
        component: '404',
    }
];

export default routes;
